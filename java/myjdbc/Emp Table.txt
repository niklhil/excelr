EMP TABLE :
-----------------------------------------------------------------------------
CREATE TABLE emp(empno INTEGER PRIMARY KEY,
		  ename  VARCHAR(25),
		  job VARCHAR(15),
		  mgr INTEGER,
		  hiredate DATE,
                  	  sal INTEGER,
		  comm INTEGER,
	                   deptno INTEGER);
